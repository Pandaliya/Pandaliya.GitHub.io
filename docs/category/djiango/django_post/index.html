@csrf_exempt
def markdown_wiki_update(request):
    # 1. 非post请求直接返回
    if request.method != 'POST':
        return JsonResponse(RES_JSON_DICT)

    except Exception as e:
        print('Reason:', e)
        return JsonResponse(RES_JSON_DICT)

    # 3. 数据更新操作
    # ...

    resp = RES_JSON_DICT
